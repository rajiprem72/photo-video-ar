      height="0.56"
      position="0 0 0"
      rotation="0 0 0">
    </a-video>
  </a-entity>

<script>
  const video = document.querySelector("#arVideo");

  document.querySelector("[mindar-image-target]").addEventListener("targetFound", () => {
    video.play();
  });

  document.querySelector("[mindar-image-target]").addEventListener("targetLost", () => {
    video.pause();
  });
